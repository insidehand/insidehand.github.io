Node & NVM

// 檢查線上版本
nvm ls-remote

// 安裝node

// 安裝特定版本
nvm install v13.14.0

// 安裝穩定版
nvm install stable

// 檢查已經安裝過的node列表
nvm ls

// 移除node
nvm uninstall v14.19.2

// 切換版本
nvm use v13.14.0
